# vuex的简单使用

  • src/store/action-types.js
export const SET_COUNT = "SET_COUNT"
export const SET_ASYNC_COUNT = "SET_ASYNC_COUNT"
  • src/store/index.js
import { createStore } from '@/vuex' //使用自己实现的vuex
import * as Types from "./action-types"
export default createStore({
  strict: true,
  state: {
    count: 0
  },
  mutations: {
    [Types.SET_COUNT](state, payload) {
      state.count += payload
    }
  },
  actions: {
    [Types.SET_ASYNC_COUNT]({ commit }, payload) {
      setTimeout(() => {
        commit(Types.SET_COUNT, payload)
      }, 0)
    }
  },
  getters: {
    double: (state) => state.count * 2
  },
  modules: {}
})

<template>
  <p>$store.state.count：{{$store.state.count}}</p>

  <p>count：{{count}}</p>
  <p>double：{{double}}</p>
  <button @click="handleClick">同步点击</button>

<script>
import { computed } from '@vue/reactivity';
import { useStore } from '@/vuex'; //使用自己实现的vuex
import * as Types from '@/store/action-types';
const useCount = (store) => {
  const handleClick = () => store.commit(Types.SET_COUNT, 1);
  const handleAysncClick = () => store.dispatch(Types.SET_ASYNC_COUNT, 1);
  return {
    handleClick,
    handleAysncClick,
  };
};
export default {
  name: 'App',
  setup() {
    const store = useStore();
    const methods = useCount(store);
    return {
      count: computed(() => store.state.count),
      double: computed(() => store.getters.double),
      ...methods,
    };
  },
};
</script>

# 实现

  • 本次主要实现state、mutations、actions、getters
    • src/vuex/index.js
import { inject,reactive } from 'vue'
export const forEachValue = (obj, fn) => {
  Object.keys(obj).forEach(key => fn(obj[key], key))
}
const storeKey = "store"
class Store {
  constructor(options) {
    const store = this;
    /**
     * state 响应式数据绑定
     */
    store._state = reactive({
      data: options.state
    })
    //计算属性绑定
    const _getters = options.getters || Object.create(null);
    store.getters = {};
    forEachValue(_getters, (fn, key) => {
      Object.defineProperty(store.getters, key, {
        /**
         * 这里不能使用computed 在vue3.1中 computed属性会在组件销毁时移除计算属性
         * vue3.2会改掉这个bug
         */
        get: () => fn(store.state),
				enumerable: true
      })
    })
    //mutation && action实现
    store._mutations = Object.create(null);
    store._actions = Object.create(null);
    const _mutations = options.mutations || Object.create(null);
    const _actions = options.actions || Object.create(null);
    forEachValue(_mutations, (mutation, key) => {
      store._mutations[key] = (payload) => {
        mutation.call(store, store.state, payload)
      }
    })
    forEachValue(_actions, (actions, key) => {
      store._actions[key] = (payload) => {
        actions.call(store, store, payload)
      }
    })
  }
  /**
   * 以下俩方法需要使用箭头函数的写法，this指向 
   * 方便使用者解构
   */
  commit = (type, payload) => {
    this._mutations[type](payload)
  }
  dispatch = (type, payload) => {
    this._actions[type](payload)
  }

  get state() { //类的访问器
    return this._state.data
  }
  /**
   * 
   * @param {*} app 注入对象
   * @param {*} injectKey 注入的新名称
   */
  install(app, injectKey) {
    //全局暴漏一个变量 暴露的是store实例
    /** `provide`给根app增加一个属性,子组件会向上查找
     */
    app.provide(injectKey || storeKey, this);
    // Vue.prototype.$store = this 兼容vue2
    app.config.globalProperties.$store = this //增添$store属性
  }
}
/**
 * @Description:创建容器
 * @param {*} options 
 * @returns store
 */
export const createStore = (options) => {
  return new Store(options)
}
export const useStore = (injectKey = null) => {
  return inject(injectKey !== null ? injectKey : storeKey)
}

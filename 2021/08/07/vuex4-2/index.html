# @/vuex/store.js

+ import { reactive } from "@vue/reactivity"
+ import { storeKey,isPromise,forEachValue } from "./utils"
import ModuleCollection from "./module/module-collection"

/**
 * 
 * @param {*} state 状态
 * @param {*} path 路径
 * @returns 路径上的状态
 */
+ const getNestedState = (state, path) => {
+  return path.reduce((state, key) => state[key], state)
+ }
/**
 * 
 * @param {*} store 实例
 * @param {*} rootState 状态
 * @param {*} path 路径
 * @param {*} root 模块
 */
+ const installModule = (store, rootState, path, module) => { //递归安装
+  let isRoot = !path.length; //如果是空数组 说明是根 否则不是
+   if (!isRoot) {
+     let parentState = path.slice(0, -1).reduce((state, key) => state[key], rootState);
+     parentState[path[path.length - 1]] = module.state
+   }
  //遍历父子关系 将子状态存放父模块
+   module.forEachChild((child, key) => {
+     installModule(store, rootState, path.concat(key), child)
+   })
  //mutations
+   module.forEachMutation((mutation, key) => {
+     const entry = store._mutations[key] || (store._mutations[key] = [])
+     entry.push(payload => {
+       mutation.call(store, getNestedState(store.state, path), payload)
+     })
+   })
  //actions
+   module.forEachAction((action, key) => {
+     const entry = store._actions[key] || (store._actions[key] = [])
+     entry.push(payload => {
+       let res = action.call(store, store, payload);
+       if (!isPromise(res)) { //处理异步
+         return Promise.resolve(res)
+       }
+       return res
+     })
+   })
  //getters
+   module.forEachGetter((getter, key) => {
+     store._wrappedGetters[key] = () => {
+       return getter(getNestedState(store.state, path)) //如果直接使用module.state 此状态不是响应式
+     }
+   })
+ }

//定义响应式数据
+ const resetStoreState = (store, state) => {
+   store._state = reactive({
+     data: state
+   })
+   const wrappedGetters = store._wrappedGetters;
+   store.getters = Object.create(null)
+   forEachValue(wrappedGetters, (getter, key) => {
+     Object.defineProperty(store.getters, key, {
+       get: () => getter(),
+       enumerable: true
+     })
+   })
+ }
export default class Store {
  constructor(options) {
    //格式化用户传进来的数据
    const store = this;
    store._modules = new ModuleCollection(options)
    //定义状态
+    const state = store._modules.root.state; //根状态
+    store._mutations = Object.create(null);
+    store._actions = Object.create(null);
+    store._wrappedGetters = Object.create(null);
    //安装状态
+    installModule(store, state, [], store._modules.root);
+    resetStoreState(store, state)
+    console.log(store)
  }
	/**
	 * 以下俩方法需要使用箭头函数的写法，this指向 
	 * 方便使用者解构
	 */
+  commit = (type, payload) => {
+    const entry = this._mutations[type] || [];
+    entry && entry.forEach(handler => handler(payload))
+  }
+  dispatch = (type, payload) => {
+    const entry = this._actions[type] || [];
+    return Promise.all(entry.map(handler => handler(payload)))
+  }
+  get state() { //类的访问器
+    return this._state.data
+  }
  /**
   * @param {*} app 注入对象
   * @param {*} injectKey 注入的新名称
   */
+  install(app, injectKey) {
    //全局暴漏一个变量 暴露的是store实例
    /** `provide`给根app增加一个属性,子组件会向上查找
     * see https://v3.cn.vuejs.org/guide/component-provide-inject.html
     */
+    app.provide(injectKey || storeKey, this);
    // Vue.prototype.$store = this 兼容vue2
+    app.config.globalProperties.$store = this //增添$store属性
+  }
}

/**
 * @Description: 遍历对象
 * @param {*} obj 对象
 * @param {*} fn 回调函数
 */
export const forEachValue = (obj, fn) => {
  Object.keys(obj).forEach(key => fn(obj[key], key))
}

# @/vuex/module/module-collection.js

import Module from "./module"
import { forEachValue } from "../utils"
export default class ModuleCollection {
  constructor(rootModule) {
    this.root = null;
    this.register(rootModule, [])
  }
  /**
   * @Description 格式化数据
   * @param {} rawModule 当前的模块
   * @param {*} path 路径
   */
  register(rawModule, path) {
    const newModule = new Module(rawModule); //存储新模块
    if (path.length == 0) { //说明是根模块
      this.root = new Module(rawModule)
    } else {
      //截取path路径 去掉最后一个
      const parent = path.slice(0, -1).reduce((module, current) => {
        return module.getChild(current);
      }, this.root); //父级
      parent.addChild(path[path.length - 1], newModule);
    }
+    if (rawModule.modules) { //说明有子模块
+      forEachValue(rawModule.modules, (rawChildModule, key) => {
+        this.register(rawChildModule, path.concat(key))
+      })
+    }
  }
}

# @/vuex/module/module.js

import { forEachValue } from "../utils";
export default class Module {
  constructor(rawModule) {
    this._raw = rawModule;
    this.state = rawModule.state;
    this._children = {};
  }
  /**
   * @Description 添加模块
   * @param {*} key 模块名
   * @param {*} module 模块
   */
  addChild(key, module) {
    this._children[key] = module
  }
  /**
   * @Description 获取模块
   * @param {*} 模块名 
   * @returns 模块
   */
  getChild(key) {
    return this._children[key]
  }
+  forEachChild(fn) {
+    forEachValue(this._children, fn)
+  }
+  forEachGetter(fn) {
+    if (this._raw.getters) {
+      forEachValue(this._raw.getters, fn)
+    }
+  }
+  forEachMutation(fn) {
+    if (this._raw.mutations) {
+      forEachValue(this._raw.mutations, fn)
+    }
+  }
+  forEachAction(fn) {
+    if (this._raw.actions) {
+      forEachValue(this._raw.actions, fn)
+    }
+  }
}

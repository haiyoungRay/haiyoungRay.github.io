# src\monitor\lib\blank-screen.js

import tracker from "../utils/tracker"
import onload from "../utils/onload";
const getSelector = (element) => {
  /**
   * 
   * @param {*} element 目标元素
   * @returns 目标元素的信息 id｜class｜标签名
   */
  var selector;
  if (element.id) {
    selector = `#${element.id}`;
  } else if (element.className && typeof element.className === 'string') {
    selector = `.${element.className}`.split(' ').filter((item) => !!item).join('.'); //多个class处理
  } else {
    selector = element.nodeName.toLowerCase();
  }
  return selector;
}
export const blankSreen = (blankElements = []) => {
  /**
   * @query {array} 预先设置目标空元素
   */
  let wrapperSelectors = ['body', 'html', '#app', '#root']
  if (Array.isArray(blankElements)) wrapperSelectors = [...wrapperSelectors, ...blankElements];
  let emptyPoints = 0;
  /**
   * @param {*} element  目标元素
   */
  const isWrapper = (element) => {
    let selector = getSelector(element);
    if (!wrapperSelectors.includes(selector)) {
      emptyPoints++;
    }
  }
  onload(() => {
    /**
     * 通过判断浏览器上的元素
     * 1.浏览器x轴的中线
     */
    let xElements, yElements;
    //screen.width  屏幕的宽度   screen.height 屏幕的高度
    //window.innerWidth 去除工具条与滚动条的窗口宽度 window.innerHeight 去除工具条与滚动条的窗口高度
    /**
     *  页面x、y轴中线上的元素 若存在 `wrapperSelectors` 上没有的目标空元素则代表非白屏
     */
    for (let i = 1; i < 10; i++) {
      if (emptyPoints > 0) break
      xElements = document.elementsFromPoint(window.innerWidth * i / 10, window.innerHeight / 2); //横轴坐标上的元素
      yElements = document.elementsFromPoint(window.innerWidth / 2, window.innerHeight * i / 10); //纵轴坐标上的元素
      isWrapper(xElements[0]);
      isWrapper(yElements[0]);
    }
    if (emptyPoints === 0) {
      /**
       *  页面x、y轴中线的中线上的元素 若存在 `wrapperSelectors` 上没有的目标空元素则代表非白屏
       */
      for (let i = 1; i < 10; i++) {
        if (emptyPoints > 0) break
        xElements = document.elementsFromPoint(window.innerWidth * i / 20, window.innerHeight / 4); //横轴坐标上的元素
        yElements = document.elementsFromPoint(window.innerWidth / 4, window.innerHeight * i / 20); //纵轴坐标上的元素
        isWrapper(xElements[0]);
        isWrapper(yElements[0]);
      }
    }
    if (emptyPoints === 0) {
      /**
       *  页面x、y轴中线的中线的中线上的元素 若存在 `wrapperSelectors` 上没有的目标空元素则代表非白屏
       */
      for (let i = 1; i < 10; i++) {
        if (emptyPoints > 0) break
        xElements = document.elementsFromPoint(window.innerWidth * i / 40, window.innerHeight / 8); //横轴坐标上的元素
        yElements = document.elementsFromPoint(window.innerWidth / 8, window.innerHeight * i / 40); //纵轴坐标上的元素
        isWrapper(xElements[0]);
        isWrapper(yElements[0]);
      }
    }

    if (emptyPoints === 0) {
      let centerElements = document.elementsFromPoint(window.innerWidth / 2, window.innerHeight / 2)
      tracker.send({
        kind: 'stability',
        type: 'blank',
        emptyPoints: "" + emptyPoints,
        screen: `${window.screen.width} x ${window.screen.height}`,
        viewPoint: `${window.innerWidth} x ${window.innerHeight}`,
        selector: getSelector(centerElements[0]),
      })
    }
  });
}

# src/monitor/utils/get-last-event.js

let lastEvent;
['click', 'pointerdown', 'touchstart', 'mousedown', 'keydown', 'mouseover'].forEach(item => {
  document.addEventListener(item, (event) => {
    lastEvent = event;
  }, {
    capture: true, //capture 控制监听器是在捕获阶段执行还是在冒泡阶段执行 捕获阶段
    passive: true //passive 的意思是顺从的，表示它不会对事件的默认行为说 no 不阻止默认事件
  });
});
export default () => {
  return lastEvent;
};

# src/monitor/utils/get-selector.js

const getSelector = (path) => {
  return path.reverse().filter((ele) => ele !== window && ele !== document).map((element) => {
    var selector;
    if (element.id) {
      return `${element.nodeName.toLowerCase()}#${element.id}`
    } else if (element.className && typeof element.className === 'string') {
      return `${element.nodeName.toLowerCase()}.${element.className}`;
    } else {
      selector = element.nodeName.toLowerCase();
    }
    return selector;
  }).join(' ');
}
export default (pathsOrTarget) => {
  if (Array.isArray(pathsOrTarget)) { //可能是一个数组
    return getSelector(pathsOrTarget);
  } else { //也可能是个对象
    var paths = [];
    var element = pathsOrTarget;
    while (element) {
      paths.push(element);
      element = element.parentNode;
    }
    return getSelector(paths);
  }
}

# src/monitor/utils/tracker.js

const userAgent = require('user-agent')
const project = "web-monitor", //项目名
  host = 'cn-beijing.log.aliyuncs.com', //域名
  logStore = 'web-store'; // 仓库名
const getExtraData = () => {
  return {
    title: document.title,
    url: location.href,
    timestamp: Date.now(),
    userAgent: userAgent.parse(navigator.userAgent).name
    // ...more
  };
}
class SendTracker {
  constructor() {
    // see: https://help.aliyun.com/document_detail/120218.html
    this.url = `http://${project}.${host}/logstores/${logStore}/track`; //上报的路径
    this.xhr = new XMLHttpRequest();
  }
  send(data = {}, callback) {
    let extraData = getExtraData();
    let logs = {
      ...extraData,
      ...data
    }
    //对象的值不能是字符串
    for (let key in logs) {
      if (typeof logs[key] === 'number') {
        logs[key] = `${logs[key]}`;
      }
    }
    let body = JSON.stringify({
      __logs__: [logs]
    });
    console.log(logs)
    this.xhr.open('POST', this.url, true);
    this.xhr.setRequestHeader("Content-Type", "application/json;charset=UTF-8"); //请求体的类型
    this.xhr.setRequestHeader('x-log-apiversion', '0.6.0'); //版本号
    this.xhr.setRequestHeader('x-log-bodyrawsize', body.length); //请求体的大小
    this.xhr.onload = () => {
      if ((this.status >= 200 && this.status <= 300) || this.status == 304) {
        callback && callback();
      }
    }
    this.xhr.onerror = (error) => {
      console.log(`error:${error}`);
    }
    this.xhr.send(body)
  }
}
export default new SendTracker()

# src/monitor/lib/js-error.js

import getLastEvent from '../utils/get-last-event';
import getSelector from "../utils/get-selector"
import tracker from "../utils/tracker"
import formatTime from "../utils/format-time"
const getLines = (stack) => {
  return stack.split('\n').slice(1).map(item => item.replace(/^\s+at\s+/g, '')).join('^');
}
export const injectJsError = () => {
  //监听全局未捕获的错误
  window.addEventListener("error", (e) => {
    let lastEvent = getLastEvent(), //最后一个交互事件
      logs_data = {}
    if (e.target && (e.target.src || e.target.href)) { //说明是一个资源
      logs_data = { //资源加载错误
        kind: 'stability', //稳定性指标
        type: 'error', //resource
        errorType: 'resourceError', //js or css资源加载错误
        filename: e.target.src || e.target.href, //加载失败的资源
        tagName: e.target.tagName, //标签名
        timeStamp: formatTime(e.timeStamp), //时间
        selector: getSelector(e.path || e.target), //选择器
      }
    } else {
      logs_data = {
        kind: 'stability', //稳定性指标
        type: 'error', //小类型 这是一个错误
        errorType: 'jsError', //js执行错误
        message: e.message, //报错信息
        filename: e.filename, //报错链接
        position: `${(e.lineno || 0)}行${(e.colno || 0)}列`, //行列号
        stack: getLines(e.error.stack), //错误堆栈
        selector: lastEvent ? getSelector(lastEvent.path || lastEvent.target) : '' //CSS选择器
      }
    }
    tracker.send(logs_data)
  }, true) // true代表在捕获阶段调用,false代表在冒泡阶段捕获,使用true或false都可以

  //当Promise 被 reject 且没有 reject 处理器的时候，会触发 unhandledrejection 事件
  window.addEventListener("unhandledrejection", (e) => {
    let lastEvent = getLastEvent(),
      message,
      line = 0,
      column = 0,
      file,
      stack, {
        reason
      } = e,
      logs_data = { //未捕获的promise错误
        kind: 'stability', //稳定性指标
        type: 'error', //jsError
        errorType: 'promiseError', //unhandledrejection
      };
    if (typeof reason === 'string') {
      message = reason;
      logs_data = {
        ...logs_data,
        ...{
          message: message,
          selector: lastEvent ? getSelector(lastEvent.path || lastEvent.target) : ''
        }
      }
    } else if (typeof reason === 'object') { //说明是一个错误对象
      if (reason.stack) {
        var matchResult = reason.stack.match(/at\s+(.+):(\d+):(\d+)/);
        if (matchResult) {
          file = matchResult[1];
          line = matchResult[2];
          column = matchResult[3];
        }
        message = reason.message;
        stack = getLines(reason.stack);
        logs_data = {
          ...logs_data,
          ...{
            message: message, //标签名
            filename: file,
            position: `${line}行${column}列`, //行列
            stack,
            selector: lastEvent ? getSelector(lastEvent.path || lastEvent.target) : ''
          }
        }
      }
    }
    tracker.send(logs_data)
  }, true)
}

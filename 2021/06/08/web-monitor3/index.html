  <script>
    const xhrMethods = (url, _query = "") => {
      let xhr = new XMLHttpRequest;
      xhr.open("GET", url, true);
      xhr.responseType = 'json';
      xhr.onload = () => {
        console.log(xhr.response)
      }
      xhr.onerror = (error) => {
        console.log(error)
      }
      xhr.send(_query)
    }

    function sendSuccess() {
      xhrMethods("/success")
    }

    function sendError() {
      xhrMethods("/error", "name=ray")
    }
  </script>
</body>

# src/monitor/lib/xhr.js

import tracker from "../utils/tracker"
export const injectXHR = () => {
  let XMLHttpRequest = window.XMLHttpRequest;
  //缓存旧的open方法
  let oldOpen = XMLHttpRequest.prototype.open;
  //重写新的open方法
  XMLHttpRequest.prototype.open = function (method, url, isAsync, username, password) {
    if (url && !url.match(/logstores/) && !url.match(/sockjs/)) { //防止死循环
      this.logData = {
        method,
        url,
        isAsync,
        username,
        password
      }
    }
    return oldOpen.apply(this, arguments);
  }
  //缓存旧的send方法
  let oldSend = XMLHttpRequest.prototype.send;
  XMLHttpRequest.prototype.send = function (body) {
    if (this.logData) { // 如果有值说明已经劫持过
      let startTime = Date.now(); //在发送之前记录一下开始的时间
      let handler = (type) => (e) => {
        let duration = Date.now() - startTime;
        let status = this.status; // 200 ｜ 500
        let statusText = this.statusText; //OK | Server | Error
        tracker.send({ //未捕获的promise错误
          kind: 'stability', //稳定性指标
          type: 'xhr', //xhr
          eventType: type, //load error abort
          pathname: this.logData.url, //接口的url地址
          status: `${status}-${statusText}`,
          duration: `${duration}`, //接口耗时
          response: this.response ? JSON.stringify(this.response) : "",
          params: body || ''
        })
      }
      this.addEventListener('load', handler('load'), false); //成功函数
      this.addEventListener('error', handler('error'), false); //失败函数
      this.addEventListener('abort', handler('abort'), false); //终止放弃函数
    }
    oldSend.apply(this, arguments);
  }
}

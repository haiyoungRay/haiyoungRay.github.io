# shadow DOM演示

  • 实现
<!DOCTYPE html>
<html lang="en">

<head>
  <meta charset="UTF-8">
  <meta http-equiv="X-UA-Compatible" content="IE=edge">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>CSS隔离方案Shadow DOM</title>
</head>

<body>
  <div>
    <p>hello word</p>
    <div id="shadow"></div>
  </div>
  <script>
    let shadowDom = document.getElementById("shadow").attachShadow({
      mode: 'closed'
    }) // closed 外界无法访问
    let pEl = document.createElement('p');
    pEl.innerHTML = 'hello ray'
    let styleEl = document.createElement('style');
    styleEl.textContent = `
      p{
        color:red
      }
    `;
    shadowDom.appendChild(styleEl)
    shadowDom.appendChild(pEl)
  </script>
</body>
</html>

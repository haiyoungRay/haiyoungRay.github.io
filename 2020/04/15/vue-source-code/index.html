# 数据处理流程(请倒叙阅读)

let observe = (data)=>{
	 // 不是对象或者是null 无需监测
    if(typeof data !== 'object' || data == null) return;
    return new Observer(data);
}
class Observer {
	constructor(data) {
		// 将用户传入的数据 通过object.defineProperty重新定义
		if(Array.isArray(data)){
			//只能拦截数组方法，数组中的每一项还需要再观测一下
			//让数组 通过链来查找我们自己的数组原型方法
			data.__proto__ = arrayMethods;
			//观测数据中的每一项
			observeArray(data);
		}else{
			this.walk(data);
		}
	}
	walk(data) {
		let keys = Object.keys(data);
		for (let i = 0; i < keys.length; i++) {
			//用户传入的key
			let key = keys[i];
			//用户传入的key
			let	value = data[key];
			defineReactive(data,key,value);
		}
	}
}
//先获取原生的数组方法
let oldArrayProtoMethods = Array.prototype;
//拷贝一个新对象，可以查找老方法，但不改变老方法
let arrayMethods = Object.create(oldArrayProtoMethods);
let methods = ['push','shift','unshift','pop','reverse','sort','splice'];
let observeArray = (inserted)=>{
	for(let i = 0;i < inserted.length;i++){
		observe(inserted[i]);
	}
}
methods.forEach(method => {
	arrayMethods[method] = function(...args){
		//函数劫持  切片编程
		let r = oldArrayProtoMethods[methods].apply(this,args);
		// todo...
		let inserted;
		switch(method){
			case 'push':
			case 'unshift':
				inserted = args;
				break;
			case 'splice':
				inserted = args.slice(2);
				break;
			default:
				break;
		}
		if(inserted) observeArray(inserted);
		return r;
	}
})
//定义响应式的数据变化
let defineReactive = (data,key,value)=>{
	//如果value依旧是一个对象或数组 需要深度观察
	observe(value); //递归观察
	Object.defineProperty(data,key,{
		get(){
			return value
		},
		set(newValue){
			//如果新值是一个对象，需继续监测新增对象
			observe(newValue);
			if(newValue !== value) value = newValue;
		}
	})
}

let proxy = (vm,source,key)=>{
	// 代理数据  vm.msg = vm._data.msg
    Object.defineProperty(vm,key,{
        get(){
            return vm[source][key];
        },
        set(newValue){
            vm[source][key] = newValue;
        }
    })
}
function initData(vm){
	// 将用户插入的数据 通过object.defineProperty重新定义
	// 用户传入的data
    let data = vm.$options.data;
	// 拷贝一份data(_data),以下数据处理均使用_data
    data = vm._data = typeof data === 'function' ? data.call(vm) : data || {};
    for(let key in data){
		//会将对vm上的取值操作和赋值操作代理给vm._data 属性
		proxy(vm,'_data',key);
    }
	// 观察数据
    observe(data);
}

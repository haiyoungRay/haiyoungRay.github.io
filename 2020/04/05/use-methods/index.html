# Object

let obj = {name:"Ray",age:20};
Object.keys(obj); //["name","age"]
Object.values(obj); //["Ray",20]
Object.entries(obj);  //[["name","Ray"],["age",20]]
let arr = [["name","Ray"],["age",20]];
Object.fromEntries(arr); //{name: "Ray", age: 20}

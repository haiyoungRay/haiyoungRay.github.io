# Object

let obj = {name:"Ray",age:20};
Object.keys(obj); //["name","age"]
Object.values(obj); //["Ray",20]
Object.entries(obj);  //[["name","Ray"],["age",20]]
let arr = [["name","Ray"],["age",20]];
Object.fromEntries(arr); //{name: "Ray", age: 20}

# Function

/**
 * @Description: 获取缓存中的字段
 * @param {boolean} isLocal  是否永久存储
 * @param {string} key  缓存中的字段名
 * @return {*} 缓存中的值
 */
export const getLocal = (isLocal,key) => {
	let winKey = isLocal ? 'localStorage' : 'sessionStorage';
  let value = window[winKey].getItem(key) || '';
  if (value.startsWith('[') || value.includes('{')) {
    return JSON.parse(value);
  } else {
    return value
  }
}
/**
 * @Description: 设置缓存
 * @param {boolean} isLocal  是否永久存储
 * @param {string} key  缓存中的字段名
 * @param {string} value  缓存中的值
 */
export const setLocal = (isLocal,key, value) => {
	let winKey = isLocal ? 'localStorage' : 'sessionStorage';
  if (typeof value == 'object') {
    value = JSON.stringify(value);
  }
  window[winKey].setItem(key, value);
}
/**
 * @Description:require.context读取文件  node环境使用
 * @param 路径
 * @param 是否读取子目录
 * @param 匹配路径的正则  (./文件下以router.js结尾的文件)
 * @return {function}
 */
const routeFiles = require.context('./', true, /\.router\.js$/),
	myRoutes = [];
routeFiles.keys().forEach(route => {
  const routerModule = routeFiles(route)
  myRoutes = [...myRoutes, ...(routerModule.default || routerModule)]
})

发布订阅

Haiyoung Ray   in beijing
Sun Apr 05 2020
  • Function
  • javascript
  • 发布方和订阅方 没有任何的关系
let e = {
	_obj:{},
	_callback:[],
	on(callback){ //订阅
		this._callback.push(callback);
	},
	emit(key,value){ //发布
		this._obj[key] = value;
		this._callback.forEach(cb=>{
			cb(this._obj);
		})
	}
}
e.on((obj)=>{
	console.log(obj);
})
e.emit("zhangsan",{address:"三元桥",phone:15278906542});
e.emit("lisi",{address:"朝阳门",phone:13256789876});

# 什么叫高阶函数

  • 如果一个函数的参数是一个函数
function callback(){
	console.log(1);
}
function a(cb){
	cb()
}
a(callback)
  • 如果一个函数的返回值是一个函数
function callback(){
	console.log(1);
}
function a(cb){
	return cb();
}
a(callback);

# 判断类型

  • typeof 无法辨别对象类型
  • constructor 可以判断当前变量的构造函数是什么(谁构造出来的当前变量)
  • instanceof 判断谁是谁的实例 使用__proto__
  • Object.prototype.toString.call 可以很精准的判断类型
function isType(content,type){
	return Object.prototype.toString.call(content) === `[object ${type}]`;
}
console.log("hello","String");
console.log(123,Number);

# 更优的写法(高阶函数)

function isType(type){
	// 将这个type保存在这个代码块中 私有化
	return function(content){
		return Object.prototype.toString.call(content) === `[object ${type}]`;
	}
}
let isString = isType("String");
isString("hello");
let isNumber = isType("Number");
isNumber(123);
// 当然我们也可以将这个方法进行封装(判断类型)
let utils = {};
['String','Number','Boolean','Null','Undefined','Symbol','Object','Array','Function'].forEach((type)=>{
	util[`is${type}`] = isType(type);
})
utils.isString(123); //false
utils.isString('hello'); //true
  • 在定义的时候，函数就决定了所在作用域
  • 一个函数不在自己所在的作用域下执行

# 切片编程

Aspect Oriented Programming(AOP)，面向切面编程，是一个比较热门的话题。AOP主要实现的目的是针对业务处理过程中的切面进行提取，它所面对的是处理过程中的某个步骤或阶段，以获得逻辑过程中各部分之间低耦合性的隔离效果。

function say(){
	console.log('say');
}
// 在说话之前，去做一些事情(说话前先刷牙)
function say(){
	console.log('刷牙');
	console.log('say');
}
//这样一来，别人调say方法也会执行执行新增的代码(刷牙)
//这显然不是我们想要的，因为这样我们改变来原函数的方法
//因此我们需要say对其扩展，而不会修改原有的say方法
function say(){
	console.log('say');
}
say.__proto__.before = function(cb){
	return ()=>{
		cb();
		this();
	}
}
// or
Function.prototype.before = function(cb){
	return ()=>{
		cb();
		this();
	}
}
let newSay = say.before(function(){
	console.log("刷牙");
});
newSay();
/*
刷牙
say
*/

# 函数的柯里化

把接收多个参数的函数变换成接收一个单一参数（最初函数的第一个参数）的函数，并返回接受剩余的参数而且返回结果的新函数的技术。其由数学家Haskell Brooks Curry提出，并以curry命名

特点

  • 可以延迟计算，即如果调用柯里化函数传入参数是不调用的，会将参数添加到数组中存储，等到没有参数传入的时候进行调用；
  • 参数复用，当在多次调用同一个函数，并且传递的参数绝大多数是相同的，那么该函数可能是一个很好的柯里化候选。

举例说明

当我们做一个求和的方法，通常我们会这样做：

let a = 0;
function sum(x){
	return a += x;
}
sum(1);  //1
sum(2);  //3

我们对上面的方法进行一点改动

function sum(x){
	return function(y){
		return x + y;
	}
}
let a = sum(0);
a(3);
a(9);

上面代码虽然简单的实现了柯里化的基本操作，但是对于参数不确定的情况是处理不了的；所以存在着函数参数的局限性；不过我们从上面的代码中基本可以理解函数柯里化的概念；就是一个函数调用的时候只允许传入一个参数，然后通过闭包返回内部函数去处理和接收剩余参数，返回的函数通过闭包的方式记住了sum的第一个参数；

我们再来把代码改造一下：

let sum = (()=>{
	let arr = [];
	return (...args)=>{
		if(args.length === 0){
			let a = 0;
			arr.forEach((item,index)=>{
				a += item;
			})
			return a;
		}else{
			arr.push(...args);
		}
	}
})();
console.log(sum(1))
console.log(sum(2))
console.log(sum())		//3

代码经过我们的改造已经实现了功能，但是这不是一个函数柯里化的完整实现，那么我们要怎么完整实现呢？下面我们来介绍一种通用的实现方式：

# 通用实现方式

/* es5 */
var currying = function(fn) {
	var arr = [];
	return function() {
		if (arguments.length === 0) {
			return fn.apply(this,arr)
		}
		Array.prototype.push.apply(arr, [].slice.call(arguments));
		//arguments.callee返回正被执行的function对象
		return arguments.callee
	}
}

function add() {
	var a = 0,arr = [].slice.call(arguments);
	for (var i = 0, l = arr.length; i < l; i++) {
		a += arr[i];
	}
	return a;
}
var s = currying(add);
s(1)(2)(3)();
s(1, 2, 3);
console.log(s());
-------------------------------------------------------------------------
/* es6 */
let currying = (fn)=> {
	let arr = [],cb = "";
	return cb = (...args)=>{
		if(args.length === 0){
			return fn(arr);
		}else{
			arr.push(...args);
		}
		return cb;
	}
}
let add = arr =>{
	return arr.reduce((prev,cur)=> prev + cur);
}
let s=currying(add);
s(1)(2)(3)();
s(1,2,3);
s(1,2)(3)(4,5,6);
console.log(s());

# 函数的反柯里化

从字面意思上来讲就是跟柯里化的意思相反；其实真正的反柯里化的作用是扩大适用范围，创建一个应用范围更广的函数。使本来只有特定对象才适用的方法，扩展到更多的对象。

在学习JS反柯里化之前，我们先学习一下动态语言的鸭子类型思想，以助于我们更好的理解：

# 动态语言鸭子类型思想（维基百科解释）:

  • 在程序设计中，鸭子类型（duck typing）是动态类型的一种风格。
  • 在这种风格中，一个对象有效的语义，不是由继承自特定的类或实现特定的接口，而是由当前方法和属性的集合决定。
  • 这个概念的名字来源于由 James Whitcomb Riley 提出的鸭子测试，“鸭子测试”可以这样表述：当看到一只鸟走起来像鸭子、游泳起来像鸭子、叫起来也像鸭子，那么这只鸟就可以被称为鸭子。

在Javascript中有很多鸭子类型的引用，比如我们在对一个变量进行赋值的时候，显然是不需要考虑对象的类型的，正是因为如此，Javascript才更加的灵活，所以Javascript是一门典型的动态类型语言；

观察者模式

Haiyoung Ray   in beijing
Sun Apr 05 2020
  • Function
  • javascript
  • observer
  • 观察者和被观察者是有关联的
  • 观察者需要将自己放在被观察者之上
  • 当被观察者状态发生变化需要通知所有的观察者
//观察者
class Observer {
	constructor(name){
		this.name = name;
	}
	//被观察者的状态发生变化会使用此方法
	update(s){
		console.log(`${this.name}:${s.name}当前的状态是${s.state}`);
	}
}
//被观察者
class Subject {
	constructor(name){
		this.name = name;
		this.state = "开心";
		this.observers = [];
	}
	// 需要将观察者放到自己的身上
	attach(o){
		this.observers.push(o);
	}
	setState(state){
		this.state = state;
		this.observers.forEach(o=>{
			o.update(this);
		})
	}
}
let baby = new Subject("小宝宝");
let parent = new Observer("爸爸");
let mother = new Observer("妈妈");
baby.attach(parent);
baby.attach(mother);
baby.setState("哭了");
baby.setState("开心");
/*
爸爸:小宝宝当前的状态是哭了
妈妈:小宝宝当前的状态是哭了
爸爸:小宝宝当前的状态是开心
妈妈:小宝宝当前的状态是开心
*/
